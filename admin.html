async function uploadToGithub() { const token = document.getElementById('gh-token').value; const fileInput = document.getElementById('file-input'); const msg = document.getElementById('commit-msg').value || "Nouveau projet"; const status = document.getElementById('status'); if (!token || !fileInput.files[0]) { status.style.color = "#ff4444"; status.innerText = "❌ Erreur : Token ou fichier manquant."; return; } const file = fileInput.files[0]; const reader = new FileReader(); const username = "LemaireThomas-ping"; const repo = "LT"; reader.onload = async function() { const content = reader.result.split(',')[1]; const path = `docs/${file.name}`; status.style.color = "#00f0ff"; status.innerText = "⏳ Phase 1 : Upload du fichier..."; try { // --- 1. UPLOAD DU FICHIER --- const fileResponse = await fetch(`https://api.github.com/repos/${username}/${repo}/contents/${path}`, { method: 'PUT', headers: { 'Authorization': `token ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ message: `Upload: ${file.name}`, content: content }) }); if (!fileResponse.ok) throw new Error("Échec upload fichier"); // --- 2. MISE À JOUR DU DATA.JSON (AUTOMATISATION) --- status.innerText = "⏳ Phase 2 : Mise à jour de la base de données..."; // On récupère le data.json actuel const getJson = await fetch(`https://api.github.com/repos/${username}/${repo}/contents/data.json`, { headers: { 'Authorization': `token ${token}` } }); const jsonData = await getJson.json(); const currentProjects = JSON.parse(atob(jsonData.content)); // On ajoute le nouveau projet à la liste const newEntry = { title: file.name.split('.')[0].replace(/-/g, ' '), description: msg, link: `https://raw.githubusercontent.com/${username}/${repo}/main/${path}`, category: detectCategory(file.name), date: new Date().toLocaleDateString() }; currentProjects.push(newEntry); // On renvoie le data.json mis à jour await fetch(`https://api.github.com/repos/${username}/${repo}/contents/data.json`, { method: 'PUT', headers: { 'Authorization': `token ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ message: `Update data.json: add ${file.name}`, content: btoa(JSON.stringify(currentProjects, null, 2)), sha: jsonData.sha // Obligatoire pour mettre à jour un fichier existant }) }); status.style.color = "#00ff00"; status.innerText = "✅ Succès total ! Fichier envoyé et base de données mise à jour."; } catch (err) { status.style.color = "#ff4444"; status.innerText = "❌ Erreur : " + err.message; } }; reader.readAsDataURL(file); } // Fonction pour deviner la catégorie function detectCategory(name) { const n = name.toLowerCase(); if (n.includes('win') || n.includes('ad')) return 'windows'; if (n.includes('linux') || n.includes('debian') || n.includes('zabbix')) return 'linux'; if (n.includes('cisco') || n.includes('vlan') || n.includes('fortinet')) return 'reseau'; return 'other'; }
